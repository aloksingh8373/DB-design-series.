 // CUSTOMERS
Table customers {
id int [pk, increment]
name varchar(100) [not null]
phone varchar(15) [not null]
email varchar(150)
address text
city varchar(50)
state varchar(50)
pincode varchar(10)
country varchar(50)
created_at timestamp
}

 // PRODUCTS
Table products {
id int [pk, increment]
name varchar(150) [not null]
product_type varchar(20) [note: 'thrifted or handmade']
category varchar(50)
brand varchar(100)
gender varchar(20) [note: 'male, female, unisex']
description text
created_at timestamp
updated_at timestamp
}

 // INVENTORY
Table inventory {
id int [pk, increment]
product_id int [not null, ref: > products.id]
size varchar(20)
color varchar(30)
quantity int [not null]
condition varchar(50) [note: 'only for thrift items']
price decimal(10,2) [not null]
created_at timestamp
updated_at timestamp
}

 // ORDERS 
Table orders {
id int [pk, increment]
customer_id int [not null, ref: > customers.id]
order_date timestamp
total_amount decimal(10,2)
order_status varchar(30) [note: 'order_received, shipped, delivered, cancelled']
}

 // ORDER ITEMS
Table order_items {
id int [pk, increment]
order_id int [not null, ref: > orders.id]
inventory_id int [not null, ref: > inventory.id]
quantity int [not null]
price_at_time decimal(10,2) [not null]
}

 // PAYMENT
Table payments {
id int [pk, increment]
order_id int [not null, ref: > orders.id]
amount decimal(10,2) [not null]
payment_status varchar(20) [note: 'paid, failed, pending']
payment_method varchar(20) [note: 'UPI, card, COD']
payment_txn_id varchar(150)
payment_date timestamp
}

 // SHIPPING
Table shipping {
id int [pk, increment]
order_id int [not null, ref: > orders.id]
shipping_address text
city varchar(50)
state varchar(50)
pincode varchar(10)
shipping_status varchar(30) [note: 'pending, shipped, delivered']
tracking_number varchar(150)
delivery_date date
}
